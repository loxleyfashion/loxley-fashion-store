const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); let birdY = 200; let velocity = 0; const gravity = 1.2; const jump = -16; let playing = true; canvas.addEventListener("click", () => { velocity = jump; }); function drawBird() { ctx.fillStyle = "#000"; ctx.beginPath(); ctx.arc(160, birdY, 12, 0, Math.PI * 2); ctx.fill(); } function loop() { if (!playing) return; ctx.clearRect(0, 0, canvas.width, canvas.height); velocity += gravity; birdY += velocity; drawBird(); if (birdY > canvas.height || birdY < 0) { endGame(); } requestAnimationFrame(loop); } function endGame() { playing = false; ctx.fillText("Loading Collection...", 85, 240); setTimeout(() => { window.location.href = "products.html"; }, 1500); } // Auto redirect after 10 sec even if no play setTimeout(() => { window.location.href = "products.html"; }, 10000); loop();
